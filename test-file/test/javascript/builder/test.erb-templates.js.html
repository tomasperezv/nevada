    'use strict';

import assert from 'assert';
import ERBTemplates from '../../../builder/erb-templates';

/**
 * @test {ERBTemplates}
 */
describe('ERBTemplates', () => {
  /**
   * @test {ERBTemplates#filter}
   */
  it('Filter individual strings', () => {
    assert.equal(ERBTemplates.filter('<%= t("cookie_consent.find_out_more") %>'), 'find_out_more');
    assert.equal(ERBTemplates.filter('<%= t("cookie_consent.cookies_explanation") %>'), 'cookies_explanation');
    assert.equal(ERBTemplates.filter('<%= link_to t("cookie_consent.find_out_more"), cookies_path, target: "_blank" %>'), '<a href="cookies_path" target="_blank">find_out_more</a>');
    assert.equal(ERBTemplates.filter('<%= NO MATCH %>'), 'NO MATCH');
  });

  it('Filter templates with multiple matches', () => {
    const template = `<p class="landing_cookie_explanation">
      <%= t("cookie_consent.cookies_explanation") %>
      <%= t("cookie_consent.cookies_extra") %>
      <%= link_to t("cookie_consent.find_out_more"), cookies_path, target: "_blank" %>
    </p>`;

    const expected = `<p class="landing_cookie_explanation">
      cookies_explanation
      cookies_extra
      <a href="cookies_path" target="_blank">find_out_more</a>
    </p>`;
    assert.equal(ERBTemplates.filter(template), expected);
  });

  it('Replacing placeholders with matches', () => {
    assert.equal(ERBTemplates._replaceMatches('[1]', ['test']), 'test');
    assert.equal(ERBTemplates._replaceMatches('<a href="[1]">[2]</a>', ['href', 'text']), '<a href="href">text</a>');
  });
});
